## 1. 概述

`DataManager` 是 Autoload 全局单例，负责加载 `data/*.json` 并提供只读数据访问。所有游戏系统**必须**通过 DataManager 获取数值，禁止直接读取 JSON 文件。

启动时自动加载所有数据文件，缓存到内存中的 Dictionary。运行期间数据只读（阶段5存档时序列化运行时状态）。

## 2. 数据加载

### 2.1 启动加载

```gdscript
func _ready() -> void:
    _load_all_data()

func _load_all_data() -> void:
    _terrains = _load_json("res://data/terrain.json")
    _units = _load_json("res://data/units.json")
    _cities = _load_json("res://data/cities.json")

    # _events = _load_json("res://data/events.json")
    # _buildings = _load_json("res://data/buildings.json")
    # _factions = _load_json("res://data/factions.json")
    # _diplomacy = _load_json("res://data/diplomacy.json")

func _load_json(path: String) -> Dictionary:
    var file := FileAccess.open(path, FileAccess.READ)
    if file == null:
        push_error("DataManager: 无法加载 %s" % path)
        return {}
    var json := JSON.new()
    var err := json.parse(file.get_as_text())
    if err != OK:
        push_error("DataManager: JSON 解析失败 %s" % path)
        return {}
    return json.data
```

## 3. 地形数据接口

```gdscript
## 获取地形定义。找不到时返回空字典并打印警告。
func get_terrain(terrain_id: String) -> Dictionary:
    if _terrains.has("terrains"):
        for t in _terrains["terrains"]:
            if t["id"] == terrain_id:
                return t
    push_warning("DataManager: 未找到地形 %s" % terrain_id)
    return {}

## 获取所有地形列表。
func get_all_terrains() -> Array:
    return _terrains.get("terrains", [])
```

返回字段：
| 字段 | 类型 | 说明 |
|:---|:---:|:---|
| `id` | String | 地形ID |
| `name` | String | 中文名 |
| `move_cost` | int | 移动力消耗（-1=不可通行） |
| `atk_mod` | float | 攻击方伤害修正 |
| `def_mod` | float | 防御方伤害修正 |
| `ambush_chance` | float | 伏击触发概率加成 |
| `ranged_accuracy_mod` | int | 远程命中率修正（百分比） |
| `cavalry_allowed` | bool | 骑兵是否可进入 |
| `is_navigable` | bool | 是否为水系（河流等） |
| `special` | String/null | 特殊标记 |

## 4. 兵种数据接口

```gdscript
## 获取基础兵种定义。
func get_unit_type(unit_id: String) -> Dictionary:
    if _units.has("unit_types"):
        for u in _units["unit_types"]:
            if u["id"] == unit_id:
                return u
    push_warning("DataManager: 未找到兵种 %s" % unit_id)
    return {}

## 获取国家特色兵种变体。
## 返回完整属性（基础属性 + stat_overrides 合并后的结果）。
func get_faction_variant(faction_id: String, base_unit_id: String) -> Dictionary:
    if _units.has("faction_variants"):
        for v in _units["faction_variants"]:
            if v["faction_id"] == faction_id and v["base_unit"] == base_unit_id:
                var base: Dictionary = get_unit_type(base_unit_id).duplicate()
                var overrides: Dictionary = v.get("stat_overrides", {})
                for key in overrides:
                    base[key] = overrides[key]
                base["variant_id"] = v["variant_id"]
                base["variant_name"] = v["variant_name"]
                base["special_description"] = v["special_description"]
                return base
    # 无变体时返回基础兵种
    return get_unit_type(base_unit_id)

## 获取某国所有特色兵种变体。
func get_faction_variants(faction_id: String) -> Array:
    var result: Array = []
    if _units.has("faction_variants"):
        for v in _units["faction_variants"]:
            if v["faction_id"] == faction_id:
                result.append(v)
    return result

## 获取所有基础兵种列表。
func get_all_unit_types() -> Array:
    return _units.get("unit_types", [])
```

```gdscript
func get_city(city_id: String) -> Dictionary:
    if _cities.has("cities"):
        for c in _cities["cities"]:
            if c["id"] == city_id:
                return c
    push_warning("DataManager: 未找到城市 %s" % city_id)
    return {}

## 获取所有城市列表。
func get_all_cities() -> Array:
    return _cities.get("cities", [])

## 获取某国所有城市。
func get_faction_cities(faction_id: String) -> Array:
    var result: Array = []
    for c in get_all_cities():
        if c["faction_id"] == faction_id:
            result.append(c)
    return result

## 获取某国首都。
func get_capital(faction_id: String) -> Dictionary:
    for c in get_all_cities():
        if c["faction_id"] == faction_id and c["is_capital"]:
            return c
    push_warning("DataManager: 未找到 %s 的首都" % faction_id)
    return {}

## 获取地图尺寸。
func get_map_size() -> Vector2i:
    return Vector2i(
        _cities.get("map_width", 20),
        _cities.get("map_height", 15)
    )
```

## 8. 辅助方法

```gdscript
## 验证数据完整性。在 _load_all_data() 后调用，确保关键字段存在。
func validate_data() -> bool:
    var valid := true
    # 检查地形
    for t in get_all_terrains():
        if not t.has("id") or not t.has("move_cost"):
            push_error("DataManager: 地形数据缺少必要字段: %s" % t)
            valid = false
    # 检查兵种
    for u in get_all_unit_types():
        if not u.has("id") or not u.has("attack"):
            push_error("DataManager: 兵种数据缺少必要字段: %s" % u)
            valid = false
    # 检查城市
    for c in get_all_cities():
        if not c.has("id") or not c.has("hex_q") or not c.has("hex_r"):
            push_error("DataManager: 城市数据缺少必要字段: %s" % c)
            valid = false
    return valid
```
